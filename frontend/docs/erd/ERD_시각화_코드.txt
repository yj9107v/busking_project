Table User {
  id bigint [pk]
  username varchar(20) [not null]
  password varchar(255) [not null]
  nickname varchar(10) [not null]
  role enum(USER, BUSKER, ADMIN) [not null]
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]
  deleted_at datetime
}

Table Busking {
  id bigint [pk]
  uuid char(36) [not null, unique]
  user_id bigint [not null, ref: > User.id]
  location_id bigint [not null, ref: > Location.id]
  date date
  start_time time
  end_time time
  description varchar(100)
  status enum(SCHEDULED, ONGOING, COMPLETED) [not null]
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]
}

Table Location {
  id bigint [pk]
  name varchar(100) [not null]
  latitude double
  longitude double
  region varchar(50) [not null]
  description text
  is_active boolean [default: true]
}

Table PromotionPost {
  id bigint [pk]
  uuid char(36) [not null, unique]
  user_id bigint [not null, ref: > User.id]
  title varchar(100) [not null]
  content text [not null]
  media_url text
  view_count int [default: 0]
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]
  category enum(ART, DANCE, MUSIC, TALK) [not null]
}

Table Review {
  id bigint [pk]
  post_id bigint [not null, ref: > PromotionPost.id]
  user_id bigint [not null, ref: > User.id]
  rating int // 1~5
  comment text
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]

Table BoardPost {
  id bigint [pk]
  uuid char(36) [not null, unique]
  user_id bigint [not null, ref: > User.id]
  title varchar(100) [not null]
  content text [not null]
  view_count int [default: 0]
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]
}

Table Comment {
  id bigint [pk]
  post_id bigint [not null, ref: > BoardPost.id]
  user_id bigint [not null, ref: > User.id]
  content text [not null]
  created_at datetime [not null]
  updated_at datetime
  is_deleted boolean [default: false]
  parent_id bigint [ref: > Comment.id]
}

Table PostView {
  id bigint [pk]
  post_id bigint [not null]
  post_type enum(PROMOTION, BOARD) [not null]
  user_id bigint [not null]
